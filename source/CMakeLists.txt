
file(GLOB_SRC
       ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp
       ${CMAKE_CURRENT_SOURCE_DIR}/src/tinyxml/*.cpp)

file(GLOB_INC
       ${CMAKE_CURRENT_SOURCE_DIR}/include/*.h
       ${CMAKE_CURRENT_SOURCE_DIR}/include/tinyxml/*.h)
       
file(GLOB_UI
        ${PROJECT_SOURCE_DIR}/ui/*.ui )
        
 file(GLOB_QRC
        ${PROJECT_SOURCE_DIR}/ressource/*.qrc )

# require proper c++
ADD_DEFINITIONS( "-pedantic -Wunused-value -O2" )
ADD_DEFINITIONS("-Wno-long-long -Wreturn-type -Werror")

IF ( WITH_QT5 )

SET(CMAKE_AUTOMOC OFF)
SET(CMAKE_INCLUDE_CURRENT_DIR ON)

INCLUDE_DIRECTORIES( BEFORE include include/tinyxml  ${Qt5Widgets_INCLUDE_DIRS} ${ROOT_INCLUDE_DIRS} )
SET(QT_LIBRARIES ${Qt5Widgets_LIBRARIES} ${Qt5Core_LIBRARIES} ${Qt5Gui_LIBRARIES} ${ROOT_LIBRARIES})
  
QT5_WRAP_CPP(MOC_SRC_FILES ${GLOB_INC} )
QT5_WRAP_UI(UI_FILES ${GLOB_UI} )
QT5_ADD_RESOURCES(QRC_FILES ${GLOB_QRC} )

ADD_LIBRARY( ${PROJECT_NAME} SHARED ${SRC_FILES} ${MOC_SRC_FILES} ${UI_FILES} )
TARGET_LINK_LIBRARIES( ${PROJECT_NAME} ${QT_LIBRARIES} )

ADD_EXECUTABLE( QtReco main/QtReco.cpp ${QRC_FILES} )
TARGET_LINK_LIBRARIES( QtReco ${PROJECT_NAME} ${PROJECT_NAME} )

QT5_USE_MODULES(QtReco Core Gui Widgets)

ELSE( WITH_QT5 )

INCLUDE_DIRECTORIES( BEFORE include include/tinyxml ${QT_INCLUDE_DIRS} ${ROOT_INCLUDE_DIRS})

SET(CMAKE_AUTOMOC OFF)
SET(CMAKE_INCLUDE_CURRENT_DIR ON)

QT4_WRAP_CPP( MOC_SRC_FILES ${INC_FILES} )
QT4_WRAP_UI( UI_FILES ${GLOB_UI} )
QT4_ADD_RESOURCES(QRC_FILES ${GLOB_QRC} )
			
ADD_LIBRARY( ${PROJECT_NAME} SHARED ${SRC_FILES} ${MOC_SRC_FILES} ${UI_FILES} )
TARGET_LINK_LIBRARIES( ${PROJECT_NAME} ${QT_LIBRARIES} ${ROOT_QTROOT_LIBRARY} ${ROOT_LIBRARIES} )

ADD_EXECUTABLE( QtReco main/QtReco.cpp ${QRC_FILES} )
TARGET_LINK_LIBRARIES( QtReco ${PROJECT_NAME} ${PROJECT_NAME} )

ENDIF ( WITH_QT5 )

# ----- Install headers -----
INSTALL_DIRECTORY( include DESTINATION . FILES_MATCHING PATTERN "*.h" )
INSTALL_DIRECTORY( include/tinyxml DESTINATION . FILES_MATCHING PATTERN "*.h" )
